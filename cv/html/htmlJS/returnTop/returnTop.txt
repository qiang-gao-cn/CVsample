    function pageScroll(){
        window.scrollBy(0,-100);
        scrolldelay = setTimeout('pageScroll()',100);
        var sTop=document.documentElement.scrollTop+document.body.scrollTop;
        if(sTop==0) clearTimeout(scrolldelay);
    }
